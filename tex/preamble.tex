\cs_new_protected:Npn \cvdata_parse_csv_line:NV #1#2
  {
    \seq_clear:N #1
    \regex_extract_all:nnN
      { (?:^|,)("(?:[^"]|"")*"|[^,]*) }
      {#2}
      \l_tmpa_seq
    \seq_map_inline:Nn \l_tmpa_seq
      {
        \tl_set:Nn \l_tmpa_tl {##1}
        \regex_replace_once:nnN { \A, } {} \l_tmpa_tl
        \regex_replace_once:nnN { \A"(.*)"\Z } { \1 } \l_tmpa_tl
        \regex_replace_all:nnN { "" } { " } \l_tmpa_tl
        \seq_put_right:NV #1 \l_tmpa_tl
      }
  }
